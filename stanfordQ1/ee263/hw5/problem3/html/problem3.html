Implement the weighted sum of objectives derived in part A

clear;
curve_smoothing;

% Create A matrix
A = zeros(n-2, n);
for i = 2:n-1
    A(i-1, i-1:i+1) = [i^4 -2*i^4 i^4];
end

mu = logspace(-8, 2);
for i = 1:length(mu);
    Ap = [1/sqrt(n) * eye(n); sqrt(mu(i)/(n-2)) .* A];
    yp = [1/sqrt(n) * f; zeros(n-2,1)];

    ghat = (Ap'*Ap)\Ap'*yp;

    c(i) = 1/n*norm(f-ghat)^2;
    d(i) = 1/(n-2)*norm(A*ghat)^2;
end

plot(c, d);
xlabel('Error in C');
ylabel('Error in D');
